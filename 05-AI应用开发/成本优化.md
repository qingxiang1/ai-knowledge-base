## 一、成本构成分析

### 1.1 成本结构拆解

```
AI应用成本构成：

直接成本（API调用）
├── 输入token费用
│   ├── 系统提示词（固定成本）
│   ├── 用户输入（可变成本）
│   ├── 历史上下文（累积成本）
│   └── 工具返回结果（附加成本）
├── 输出token费用
│   ├── 通常比输入贵2-4倍
│   ├── 受max_tokens限制
│   └── 与任务复杂度正相关
└── 调用次数
    ├── 用户交互频次
    ├── 重试机制触发
    └── 后台任务调度

间接成本
├── 基础设施
│   ├── 服务器/容器运行
│   ├── 带宽和CDN
│   └── 数据库和缓存
├── 开发运维
│   ├── 模型评估和测试
│   ├── 提示词迭代优化
│   └── 监控和告警
└── 数据成本
    ├── 数据标注和清洗
    ├── 向量数据库存储
    └── 日志和审计存储

成本公式：
总成本 = Σ(输入tokens × 输入单价 + 输出tokens × 输出单价) × 调用次数 + 基础设施成本
```

### 2.1 分层模型架构

```python
"""
分层模型架构：根据任务复杂度选择不同模型
"""

from enum import Enum
from typing import Optional

class TaskComplexity(Enum):
    """任务复杂度分级"""
    SIMPLE = "simple"       # 简单任务：分类、提取、短回答
    STANDARD = "standard"   # 标准任务：对话、摘要、翻译
    COMPLEX = "complex"     # 复杂任务：推理、分析、代码
    CREATIVE = "creative"   # 创意任务：写作、设计、头脑风暴

class ModelRouter:
    """模型路由器 - 根据任务选择最优模型"""

    # 模型配置：模型名 -> (输入单价/1M, 输出单价/1M, 质量分)
    MODELS = {
        "gpt-4o-mini": (0.15, 0.60, 75),
        "gpt-4o": (2.50, 10.00, 95),
        "claude-3-haiku": (0.25, 1.25, 78),
        "claude-3-5-sonnet": (3.00, 15.00, 96),
        "gemini-1.5-flash": (0.075, 0.30, 80),
        "gemini-1.5-pro": (1.25, 5.00, 94),
        "deepseek-chat": (0.14, 0.28, 88),
        "deepseek-reasoner": (0.55, 2.19, 93),
    }

    # 任务-模型映射
    TASK_MODELS = {
        TaskComplexity.SIMPLE: [
            "gemini-1.5-flash",
            "gpt-4o-mini",
            "deepseek-chat"
        ],
        TaskComplexity.STANDARD: [
            "deepseek-chat",
            "gpt-4o-mini",
            "gemini-1.5-flash"
        ],
        TaskComplexity.COMPLEX: [
            "deepseek-reasoner",
            "claude-3-5-sonnet",
            "gpt-4o"
        ],
        TaskComplexity.CREATIVE: [
            "claude-3-5-sonnet",
            "gpt-4o",
            "gemini-1.5-pro"
        ]
    }

    def __init__(self, quality_threshold: float = 0.8, budget_priority: bool = False):
        """
        初始化路由器

        Args:
            quality_threshold: 最低质量要求 (0-1)
            budget_priority: 是否优先成本
        """
        self.quality_threshold = quality_threshold
        self.budget_priority = budget_priority

    def select_model(
        self,
        task_complexity: TaskComplexity,
        expected_input_tokens: int = 1000,
        expected_output_tokens: int = 500,
        preferred_provider: Optional[str] = None
    ) -> dict:
        """
        选择最优模型

        Returns:
            {
                "model": 模型名,
                "estimated_cost": 预估成本,
                "quality_score": 质量分,
                "reason": 选择理由
            }
        """
        candidates = self.TASK_MODELS[task_complexity]

        best_choice = None
        best_score = float('inf') if self.budget_priority else 0

        for model in candidates:
            if model not in self.MODELS:
                continue

            input_price, output_price, quality = self.MODELS[model]

            # 质量过滤
            normalized_quality = quality / 100
            if normalized_quality < self.quality_threshold:
                continue

            # 计算预估成本
            cost = (expected_input_tokens / 1e6 * input_price +
                   expected_output_tokens / 1e6 * output_price)

            # 计算综合得分
            if self.budget_priority:
                # 成本优先：成本越低越好，但质量不能太低
                score = cost / normalized_quality
                if score < best_score:
                    best_score = score
                    best_choice = {
                        "model": model,
                        "estimated_cost": round(cost, 6),
                        "quality_score": quality,
                        "reason": f"成本优先：${cost:.6f}，质量分{quality}"
                    }
            else:
                # 质量优先：质量越高越好，但成本不能太高
                score = normalized_quality / (cost + 0.000001)
                if score > best_score:
                    best_score = score
                    best_choice = {
                        "model": model,
                        "estimated_cost": round(cost, 6),
                        "quality_score": quality,
                        "reason": f"质量优先：质量分{quality}，成本${cost:.6f}"
                    }

        return best_choice or {
            "model": "gpt-4o-mini",
            "estimated_cost": 0,
            "quality_score": 75,
            "reason": "默认选择"
        }

    def cascade_call(
        self,
        messages: list,
        task_complexity: TaskComplexity = TaskComplexity.STANDARD,
        fallback_threshold: float = 0.7
    ) -> dict:
        """
        级联调用：先用便宜模型，不满意再用贵模型

        策略：
        1. 先用低成本模型尝试
        2. 评估输出质量（通过置信度或规则判断）
        3. 质量不达标时升级到更强模型
        """
        # 第一级：便宜模型
        cheap_model = self.select_model(
            task_complexity,
            budget_priority=True
        )

        # 模拟调用（实际实现中调用API）
        # response = call_api(cheap_model["model"], messages)
        # quality_score = evaluate_quality(response)

        # 如果质量不达标，升级模型
        # if quality_score < fallback_threshold:
        #     expensive_model = self.select_model(
        #         task_complexity,
        #         budget_priority=False
        #     )
        #     response = call_api(expensive_model["model"], messages)
        #     return {"model": expensive_model["model"], "response": response, "cascade": True}

        return {
            "model": cheap_model["model"],
            "cascade": False,
            "cost_saved": True
        }

# 使用示例

# 简单任务

    TaskComplexity.SIMPLE,
    expected_input_tokens=500,
    expected_output_tokens=200
)
print(f"简单任务推荐: {result['model']}, 预估成本: ${result['estimated_cost']}")

    expected_output_tokens=1500
)
```

### 2.2 模型降级策略

```python
"""
模型降级策略：在高负载或预算紧张时自动降级
"""

import time
from collections import deque

class AdaptiveModelSelector:
    """自适应模型选择器"""

    def __init__(
        self,
        daily_budget_usd: float = 100.0,
        budget_warning_threshold: float = 0.8,
        high_load_threshold: int = 100  # QPS
    ):
        self.daily_budget = daily_budget_usd
        self.warning_threshold = budget_warning_threshold
        self.high_load_threshold = high_load_threshold

        self.daily_spend = 0.0
        self.request_times = deque(maxlen=1000)  # 最近1000次请求时间
        self.model_usage = {}  # 模型使用统计

    def get_current_qps(self) -> float:
        """计算当前QPS"""
        now = time.time()
        recent = [t for t in self.request_times if now - t < 60]
        return len(recent) / 60.0

    def get_budget_usage(self) -> float:
        """获取预算使用率"""
        return self.daily_spend / self.daily_budget

    def select_model(self, default_model: str, fallback_model: str) -> str:
        """
        根据当前状况选择模型

        降级触发条件：
        1. 预算使用率超过阈值
        2. QPS超过高负载阈值
        3. 特定时间段（如高峰期）
        """
        budget_usage = self.get_budget_usage()
        current_qps = self.get_current_qps()

        # 检查是否需要降级
        should_downgrade = (
            budget_usage > self.warning_threshold or
            current_qps > self.high_load_threshold
        )

        if should_downgrade:
            return fallback_model

        return default_model

    def record_usage(self, model: str, cost_usd: float):
        """记录使用"""
        self.daily_spend += cost_usd
        self.request_times.append(time.time())
        self.model_usage[model] = self.model_usage.get(model, 0) + 1

    def get_stats(self) -> dict:
        """获取统计信息"""
        return {
            "daily_spend": round(self.daily_spend, 4),
            "budget_remaining": round(self.daily_budget - self.daily_spend, 4),
            "budget_usage_percent": round(self.get_budget_usage() * 100, 2),
            "current_qps": round(self.get_current_qps(), 2),
            "model_distribution": self.model_usage
        }

# 使用示例
selector = AdaptiveModelSelector(daily_budget_usd=50.0)

# 模拟高负载场景
for _ in range(200):
    selector.record_usage("gpt-4o", 0.01)

model = selector.select_model("gpt-4o", "gpt-4o-mini")
print(f"当前预算使用率: {selector.get_budget_usage():.1%}")
print(f"选择模型: {model}")
```

### 3.1 提示词缓存

```python
"""
提示词缓存：缓存常见查询的响应，减少API调用
"""

import hashlib
import json
import time
from typing import Optional, Dict, Any

class PromptCache:
    """提示词缓存系统"""

    def __init__(self, ttl_seconds: int = 3600, max_size: int = 10000):
        """
        初始化缓存

        Args:
            ttl_seconds: 缓存过期时间
            max_size: 最大缓存条目数
        """
        self.ttl = ttl_seconds
        self.max_size = max_size
        self.cache: Dict[str, Dict] = {}
        self.hit_count = 0
        self.miss_count = 0

    def _generate_key(self, messages: list, model: str, **kwargs) -> str:
        """生成缓存键"""
        # 对消息和参数进行哈希
        cache_data = {
            "messages": messages,
            "model": model,
            "params": kwargs
        }
        json_str = json.dumps(cache_data, sort_keys=True, ensure_ascii=False)
        return hashlib.sha256(json_str.encode()).hexdigest()

    def get(
        self,
        messages: list,
        model: str,
        **kwargs
    ) -> Optional[str]:
        """获取缓存"""
        key = self._generate_key(messages, model, **kwargs)

        if key in self.cache:
            entry = self.cache[key]

            # 检查是否过期
            if time.time() - entry["timestamp"] < self.ttl:
                self.hit_count += 1
                return entry["response"]
            else:
                # 过期删除
                del self.cache[key]

        self.miss_count += 1
        return None

    def set(
        self,
        messages: list,
        model: str,
        response: str,
        **kwargs
    ):
        """设置缓存"""
        # 清理过期条目
        self._cleanup()

        # 如果缓存满了，删除最旧的
        if len(self.cache) >= self.max_size:
            oldest_key = min(self.cache.keys(), key=lambda k: self.cache[k]["timestamp"])
            del self.cache[oldest_key]

        key = self._generate_key(messages, model, **kwargs)
        self.cache[key] = {
            "response": response,
            "timestamp": time.time()
        }

    def _cleanup(self):
        """清理过期缓存"""
        now = time.time()
        expired = [
            key for key, entry in self.cache.items()
            if now - entry["timestamp"] > self.ttl
        ]
        for key in expired:
            del self.cache[key]

    def get_stats(self) -> dict:
        """获取缓存统计"""
        total = self.hit_count + self.miss_count
        hit_rate = self.hit_count / total if total > 0 else 0

        return {
            "cache_size": len(self.cache),
            "hit_count": self.hit_count,
            "miss_count": self.miss_count,
            "hit_rate": round(hit_rate, 4),
            "estimated_savings": f"${self.hit_count * 0.01:.2f}"  # 假设每次调用$0.01
        }

# 语义缓存（基于嵌入向量）
class SemanticCache:
    """语义缓存：缓存语义相似的查询"""

    def __init__(self, similarity_threshold: float = 0.95):
        self.threshold = similarity_threshold
        self.entries = []  # [(embedding, response, timestamp)]

    def find_similar(
        self,
        query_embedding: list,
        top_k: int = 1
    ) -> Optional[str]:
        """
        查找语义相似的缓存

        需要配合向量数据库使用，如：
        - Redis Vector
        - Pinecone
        - Milvus
        - pgvector
        """
        # 简化实现：线性搜索
        # 实际应使用向量数据库
        best_similarity = 0
        best_response = None

        for entry in self.entries:
            similarity = self._cosine_similarity(query_embedding, entry[0])
            if similarity > best_similarity:
                best_similarity = similarity
                best_response = entry[1]

        if best_similarity >= self.threshold:
            return best_response

        return None

    def _cosine_similarity(self, a: list, b: list) -> float:
        """计算余弦相似度"""
        dot = sum(x * y for x, y in zip(a, b))
        norm_a = sum(x * x for x in a) ** 0.5
        norm_b = sum(x * x for x in b) ** 0.5
        return dot / (norm_a * norm_b) if norm_a and norm_b else 0
```

```python
"""
常见缓存模式
"""

class CachePatterns:
    """缓存设计模式"""

    @staticmethod
    def exact_match_cache(query: str, cache: dict) -> Optional[str]:
        """
        精确匹配缓存

        适用：FAQ、固定知识查询
        示例：
        - "公司地址是什么" → "北京市..."
        - "如何重置密码" → "步骤1..."
        """
        return cache.get(query.strip().lower())

    @staticmethod
    def template_cache(query: str, templates: dict) -> Optional[str]:
        """
        模板缓存

        适用：结构化查询
        示例：
        - "查询订单 {order_id}" → 预定义模板
        - "翻译 {text} 到 {language}" → 预定义模板
        """
        for pattern, handler in templates.items():
            if pattern in query:
                return handler(query)
        return None

    @staticmethod
    def session_cache(session_id: str, query: str, session_store: dict) -> Optional[str]:
        """
        会话缓存

        适用：多轮对话中的重复确认
        示例：
        - 用户问"还有什么" → 基于上文生成
        - 用户问"详细说明" → 基于上文展开
        """
        session = session_store.get(session_id, {})

        # 检查是否是跟进问题
        follow_up_keywords = ["还有什么", "详细说明", "举个例子", "为什么"]
        if any(kw in query for kw in follow_up_keywords):
            # 基于会话上下文生成，无需调用API
            context = session.get("last_response", "")
            return f"基于上文：{context[:100]}..."

        return None
```

## 四、批处理与异步化

```python
"""
批处理优化：合并多个请求，减少调用次数
"""

import asyncio
from typing import List, Callable
from dataclasses import dataclass

@dataclass
class BatchRequest:
    """批处理请求"""
    id: str
    messages: list
    callback: Callable

class BatchProcessor:
    """批处理器"""

    def __init__(self, batch_size: int = 10, max_wait_ms: int = 100):
        self.batch_size = batch_size
        self.max_wait_ms = max_wait_ms
        self.pending_requests: List[BatchRequest] = []
        self.processing = False

    async def submit(self, request: BatchRequest) -> str:
        """提交请求到批处理队列"""
        self.pending_requests.append(request)

        # 触发批处理
        if len(self.pending_requests) >= self.batch_size:
            return await self._process_batch()

        # 等待一段时间，看是否有更多请求
        await asyncio.sleep(self.max_wait_ms / 1000)

        if self.pending_requests:
            return await self._process_batch()

    async def _process_batch(self) -> List[str]:
        """处理一批请求"""
        if not self.pending_requests:
            return []

        batch = self.pending_requests[:self.batch_size]
        self.pending_requests = self.pending_requests[self.batch_size:]

        # 合并请求（如果API支持）
        # 或使用 asyncio.gather 并发调用
        tasks = [self._call_api(req) for req in batch]
        results = await asyncio.gather(*tasks)

        # 回调处理
        for req, result in zip(batch, results):
            if req.callback:
                req.callback(result)

        return results

    async def _call_api(self, request: BatchRequest) -> str:
        """调用API（模拟）"""

# 异步队列处理
class AsyncQueue:
    """异步队列处理器"""

    def __init__(self, max_concurrent: int = 5):
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.results = {}

    async def process_items(
        self,
        items: List[dict],
        processor: Callable
    ) -> List[dict]:
        """
        并发处理多个项目，但限制并发数

        适用场景：
        - 批量文档处理
        - 批量数据分析
        - 批量内容生成
        """
        async def process_with_limit(item):
            async with self.semaphore:
                return await processor(item)

        tasks = [process_with_limit(item) for item in items]
        return await asyncio.gather(*tasks)

# 使用示例
async def main():
    processor = BatchProcessor(batch_size=5)

    # 提交多个请求
    requests = [
        BatchRequest(f"req_{i}", [{"role": "user", "content": f"问题{i}"}], None)
        for i in range(10)
    ]

    tasks = [processor.submit(req) for req in requests]
    results = await asyncio.gather(*tasks)
    print(f"处理了 {len(results)} 个请求")
```

## 五、提示词优化降本

```python
"""
提示词优化：减少token消耗，提升输出质量
"""

class PromptOptimizer:
    """提示词优化器"""

    @staticmethod
    def compress_system_prompt(prompt: str) -> str:
        """
        压缩系统提示词

        策略：
        1. 去除冗余空格和换行
        2. 使用缩写
        3. 删除不必要的修饰词
        """
        # 去除多余空白
        compressed = " ".join(prompt.split())

        # 常见替换
        replacements = {
            "请不要": "勿",
            "非常重要": "关键",
            "详细说明": "详述",
            "例如": "如",
            "等等": "等",
        }

        for old, new in replacements.items():
            compressed = compressed.replace(old, new)

        return compressed

    @staticmethod
    def few_shot_optimization(
        examples: list,
        max_examples: int = 3,
        max_example_length: int = 200
    ) -> list:
        """
        优化few-shot示例

        策略：
        1. 限制示例数量
        2. 截断过长示例
        3. 选择最具代表性的示例
        """
        # 选择最短的示例（通常更通用）
        sorted_examples = sorted(examples, key=lambda x: len(x["input"]))
        selected = sorted_examples[:max_examples]

        # 截断过长示例
        for ex in selected:
            if len(ex["input"]) > max_example_length:
                ex["input"] = ex["input"][:max_example_length] + "..."
            if len(ex["output"]) > max_example_length:
                ex["output"] = ex["output"][:max_example_length] + "..."

        return selected

    @staticmethod
    def context_pruning(
        messages: list,
        max_context_tokens: int = 4000
    ) -> list:
        """
        上下文修剪

        策略：
        1. 保留系统消息
        2. 保留最近的用户-助手对话
        3. 摘要旧对话
        """
        # 分离系统消息
        system_msgs = [m for m in messages if m["role"] == "system"]
        other_msgs = [m for m in messages if m["role"] != "system"]

        # 如果上下文不长，直接返回
        total_length = sum(len(m["content"]) for m in messages)
        if total_length < max_context_tokens * 4:  # 粗略估算
            return messages

        # 保留最近4轮对话
        recent_msgs = other_msgs[-8:]  # 4轮 = 8条消息

        # 旧对话生成摘要
        old_msgs = other_msgs[:-8]
        if old_msgs:
            summary = f"[历史对话摘要] 共{len(old_msgs)//2}轮对话"
            system_msgs.append({"role": "system", "content": summary})

        return system_msgs + recent_msgs

# 提示词优化效果示例
optimizer = PromptOptimizer()

# 压缩前
long_prompt = """
你是一个非常有帮助的AI助手。你的任务是帮助用户解决各种问题。
请确保你的回答详细、准确、易于理解。如果用户的问题不清楚，
请礼貌地请用户澄清。不要使用任何不当语言。非常重要的一点是，
你必须始终保持专业和友好的态度。
"""

# 压缩后
compressed = optimizer.compress_system_prompt(long_prompt)
print(f"压缩前: {len(long_prompt)} 字符")
print(f"压缩后: {len(compressed)} 字符")
print(f"节省: {(1 - len(compressed)/len(long_prompt)):.1%}")
```

## 八、常见误区补充

| 误区                 | 问题                     | 正确做法                           |
| -------------------- | ------------------------ | ---------------------------------- |
| 只换便宜模型         | 质量下滑导致隐性成本更高 | 联动看成功率、人工接管率和满意度   |
| 只优化 prompt        | 只能做局部降本           | 同时做模型路由、缓存和异步化       |
| 不做预算上限         | 业务爆量时成本失控       | 设定请求级、用户级、日级预算阈值   |
| 忽略工具和上下文成本 | Agent 场景成本被低估     | 将工具调用和历史上下文纳入成本模型 |
| 没有质量门禁         | 降本后体验明显下降       | 为降本动作设置效果门槛和回滚条件   |
