## 二、数据接入与处理

### 2.1 多源数据接入

```python
"""
企业知识库数据接入模块

支持多种数据源的标准化接入
"""

from abc import ABC, abstractmethod
from typing import List, Dict, Iterator
from dataclasses import dataclass
from datetime import datetime
import hashlib

@dataclass
class Document:
    """标准化文档结构"""
    id: str
    content: str
    title: str
    source: str  # 数据来源
    doc_type: str  # 文档类型
    metadata: Dict
    created_at: datetime
    updated_at: datetime
    checksum: str  # 内容校验和

class DataSourceConnector(ABC):
    """数据源连接器基类"""
    
    @abstractmethod
    def connect(self):
        """建立连接"""
        pass
    
    @abstractmethod
    def fetch_documents(self) -> Iterator[Document]:
        """获取文档流"""
        pass
    
    @abstractmethod
    def get_changes(self, since: datetime) -> Iterator[Document]:
        """获取变更文档"""
        pass

class FileSystemConnector(DataSourceConnector):
    """文件系统连接器"""
    
    def __init__(
        self,
        root_path: str,
        supported_extensions: List[str] = None
    ):
        """
        初始化文件系统连接器
        
        Args:
            root_path: 根目录路径
            supported_extensions: 支持的文件扩展名
        """
        self.root_path = root_path
        self.supported_extensions = supported_extensions or [
            ".pdf", ".docx", ".txt", ".md",
            ".html", ".pptx", ".xlsx"
        ]
    
    def connect(self):
        """验证路径可访问"""
        import os
        if not os.path.exists(self.root_path):
            raise FileNotFoundError(f"路径不存在: {self.root_path}")
    
    def fetch_documents(self) -> Iterator[Document]:
        """
        遍历文件系统获取文档
        """
        import os
        
        for root, dirs, files in os.walk(self.root_path):
            for filename in files:
                ext = os.path.splitext(filename)[1].lower()
                if ext not in self.supported_extensions:
                    continue
                
                filepath = os.path.join(root, filename)
                
                try:
                    doc = self._parse_file(filepath)
                    if doc:
                        yield doc
                except Exception as e:
                    print(f"解析文件失败 {filepath}: {e}")
    
    def _parse_file(self, filepath: str) -> Document:
        """解析单个文件"""
        import os
        
        # 根据扩展名选择解析器
        ext = os.path.splitext(filepath)[1].lower()
        
        if ext == ".pdf":
            content = self._parse_pdf(filepath)
        elif ext == ".docx":
            content = self._parse_docx(filepath)
        elif ext in [".txt", ".md"]:
            content = self._parse_text(filepath)
        else:
            content = ""
        
        if not content:
            return None
        
        # 计算校验和
        checksum = hashlib.md5(content.encode()).hexdigest()
        
        stat = os.stat(filepath)
        
        return Document(
            id=hashlib.md5(filepath.encode()).hexdigest(),
            content=content,
            title=os.path.basename(filepath),
            source="filesystem",
            doc_type=ext.lstrip("."),
            metadata={
                "filepath": filepath,
                "size": stat.st_size,
                "extension": ext
            },
            created_at=datetime.fromtimestamp(stat.st_ctime),
            updated_at=datetime.fromtimestamp(stat.st_mtime),
            checksum=checksum
        )
    
    def _parse_pdf(self, filepath: str) -> str:
        """解析 PDF"""
        try:
            import PyPDF2
            with open(filepath, 'rb') as f:
                reader = PyPDF2.PdfReader(f)
                text = ""
                for page in reader.pages:
                    text += page.extract_text() + "\n"
                return text
        except ImportError:
            print("请安装 PyPDF2: pip install PyPDF2")
            return ""
    
    def _parse_docx(self, filepath: str) -> str:
        """解析 Word"""
        try:
            import docx
            doc = docx.Document(filepath)
            return "\n".join([para.text for para in doc.paragraphs])
        except ImportError:
            print("请安装 python-docx: pip install python-docx")
            return ""
    
    def _parse_text(self, filepath: str) -> str:
        """解析文本文件"""
        with open(filepath, 'r', encoding='utf-8') as f:
            return f.read()
    
    def get_changes(self, since: datetime) -> Iterator[Document]:
        """
        获取变更文件
        
        基于修改时间判断
        """
        import os
        
        for doc in self.fetch_documents():
            if doc.updated_at > since:
                yield doc

class DatabaseConnector(DataSourceConnector):
    """数据库连接器"""
    
    def __init__(
        self,
        connection_string: str,
        query: str,
        id_column: str = "id",
        content_column: str = "content"
    ):
        """
        初始化数据库连接器
        
        Args:
            connection_string: 数据库连接字符串
            query: 查询 SQL
            id_column: ID 列名
            content_column: 内容列名
        """
        self.connection_string = connection_string
        self.query = query
        self.id_column = id_column
        self.content_column = content_column
        self.engine = None
    
    def connect(self):
        """建立数据库连接"""
        from sqlalchemy import create_engine
        self.engine = create_engine(self.connection_string)
    
    def fetch_documents(self) -> Iterator[Document]:
        """从数据库获取文档"""
        import pandas as pd
        
        df = pd.read_sql(self.query, self.engine)
        
        for _, row in df.iterrows():
            content = str(row[self.content_column])
            checksum = hashlib.md5(content.encode()).hexdigest()
            
            yield Document(
                id=str(row[self.id_column]),
                content=content,
                title=str(row.get("title", "")),
                source="database",
                doc_type="database_record",
                metadata=row.to_dict(),
                created_at=row.get("created_at", datetime.now()),
                updated_at=row.get("updated_at", datetime.now()),
                checksum=checksum
            )
    
    def get_changes(self, since: datetime) -> Iterator[Document]:
        """获取变更记录"""
        # 在查询中添加时间过滤条件
        query = f"{self.query} WHERE updated_at > '{since}'"
        
        import pandas as pd
        df = pd.read_sql(query, self.engine)
        
        for _, row in df.iterrows():
            content = str(row[self.content_column])
            checksum = hashlib.md5(content.encode()).hexdigest()
            
            yield Document(
                id=str(row[self.id_column]),
                content=content,
                title=str(row.get("title", "")),
                source="database",
                doc_type="database_record",
                metadata=row.to_dict(),
                created_at=row.get("created_at", datetime.now()),
                updated_at=row.get("updated_at", datetime.now()),
                checksum=checksum
            )

# 数据源管理器
class DataSourceManager:
    """数据源管理器"""
    
    def __init__(self):
        self.connectors: Dict[str, DataSourceConnector] = {}
    
    def register_connector(
        self,
        name: str,
        connector: DataSourceConnector
    ):
        """注册数据源连接器"""
        self.connectors[name] = connector
        connector.connect()
        print(f"已注册数据源: {name}")
    
    def fetch_all(self) -> Iterator[Document]:
        """获取所有数据源的文档"""
        for name, connector in self.connectors.items():
            print(f"正在获取数据源: {name}")
            for doc in connector.fetch_documents():
                yield doc
    
    def get_all_changes(self, since: datetime) -> Iterator[Document]:
        """获取所有数据源的变更"""
        for name, connector in self.connectors.items():
            for doc in connector.get_changes(since):
                yield doc

# 使用示例
"""
manager = DataSourceManager()

# 注册文件系统数据源
fs_connector = FileSystemConnector("/path/to/documents")
manager.register_connector("documents", fs_connector)

# 注册数据库数据源
db_connector = DatabaseConnector(
    connection_string="postgresql://user:pass@localhost/db",
    query="SELECT id, title, content, created_at FROM knowledge_base"
)
manager.register_connector("database", db_connector)

# 获取所有文档
for doc in manager.fetch_all():
    print(f"[{doc.source}] {doc.title}: {len(doc.content)} 字符")
"""
```

### 2.2 文档处理流水线

```python
"""
文档处理流水线

将原始文档转换为知识库可用的结构化数据
"""

from typing import List, Callable
import re

class DocumentProcessor:
    """文档处理器"""
    
    def __init__(self):
        """初始化文档处理器"""
        self.processors: List[Callable] = []
    
    def add_processor(self, processor: Callable):
        """添加处理步骤"""
        self.processors.append(processor)
    
    def process(self, doc: Document) -> Document:
        """
        处理文档
        
        按顺序执行所有处理步骤
        """
        for processor in self.processors:
            doc = processor(doc)
        return doc

# 常用处理步骤
class TextCleaners:
    """文本清洗工具"""
    
    @staticmethod
    def remove_extra_whitespace(doc: Document) -> Document:
        """去除多余空白"""
        doc.content = re.sub(r'\s+', ' ', doc.content)
        return doc
    
    @staticmethod
    def remove_html_tags(doc: Document) -> Document:
        """去除 HTML 标签"""
        doc.content = re.sub(r'<[^>]+>', '', doc.content)
        return doc
    
    @staticmethod
    def normalize_unicode(doc: Document) -> Document:
        """统一 Unicode 编码"""
        import unicodedata
        doc.content = unicodedata.normalize('NFKC', doc.content)
        return doc
    
    @staticmethod
    def remove_control_chars(doc: Document) -> Document:
        """去除控制字符"""
        doc.content = ''.join(
            char for char in doc.content
            if ord(char) >= 32 or char in '\n\t'
        )
        return doc

class ContentEnrichers:
    """内容增强工具"""
    
    @staticmethod
    def extract_keywords(doc: Document) -> Document:
        """提取关键词"""
        # 简化的关键词提取
        words = doc.content.split()
        word_freq = {}
        for word in words:
            if len(word) > 2:
                word_freq[word] = word_freq.get(word, 0) + 1
        
        # 取频率最高的 10 个词
        keywords = sorted(word_freq.items(), key=lambda x: x[1], reverse=True)[:10]
        doc.metadata["keywords"] = [k for k, _ in keywords]
        
        return doc
    
    @staticmethod
    def generate_summary(doc: Document, max_length: int = 200) -> Document:
        """生成摘要"""
        # 简化的摘要生成：取前 max_length 个字符
        doc.metadata["summary"] = doc.content[:max_length] + "..."
        return doc
    
    @staticmethod
    def detect_language(doc: Document) -> Document:
        """检测语言"""
        try:
            import langdetect
            doc.metadata["language"] = langdetect.detect(doc.content)
        except ImportError:
            doc.metadata["language"] = "unknown"
        return doc

# 构建处理流水线
def create_default_pipeline() -> DocumentProcessor:
    """创建默认处理流水线"""
    processor = DocumentProcessor()
    
    # 文本清洗
    processor.add_processor(TextCleaners.remove_control_chars)
    processor.add_processor(TextCleaners.remove_html_tags)
    processor.add_processor(TextCleaners.normalize_unicode)
    processor.add_processor(TextCleaners.remove_extra_whitespace)
    
    # 内容增强
    processor.add_processor(ContentEnrichers.detect_language)
    processor.add_processor(ContentEnrichers.extract_keywords)
    processor.add_processor(ContentEnrichers.generate_summary)
    
    return processor

# 使用示例
"""
processor = create_default_pipeline()

# 处理文档
for doc in manager.fetch_all():
    processed_doc = processor.process(doc)
    print(f"处理完成: {processed_doc.title}")
    print(f"语言: {processed_doc.metadata.get('language')}")
    print(f"关键词: {processed_doc.metadata.get('keywords', [])}")
"""
```

### 3.1 基于 RBAC 的权限控制

```python
"""
企业知识库权限控制模块

实现基于角色的访问控制（RBAC）
"""

from enum import Enum
from typing import Set, Optional
from functools import wraps

class Permission(Enum):
    """权限枚举"""
    READ = "read"
    WRITE = "write"
    DELETE = "delete"
    ADMIN = "admin"
    SEARCH = "search"
    DOWNLOAD = "download"

class Role:
    """角色定义"""
    
    def __init__(self, name: str, permissions: Set[Permission]):
        """
        初始化角色
        
        Args:
            name: 角色名称
            permissions: 权限集合
        """
        self.name = name
        self.permissions = permissions
    
    def has_permission(self, permission: Permission) -> bool:
        """检查是否有指定权限"""
        return permission in self.permissions

# 预定义角色
ROLES = {
    "admin": Role("admin", {
        Permission.READ, Permission.WRITE,
        Permission.DELETE, Permission.ADMIN,
        Permission.SEARCH, Permission.DOWNLOAD
    }),
    "editor": Role("editor", {
        Permission.READ, Permission.WRITE,
        Permission.SEARCH, Permission.DOWNLOAD
    }),
    "viewer": Role("viewer", {
        Permission.READ, Permission.SEARCH
    }),
    "guest": Role("guest", {
        Permission.SEARCH
    })
}

class AccessControl:
    """访问控制器"""
    
    def __init__(self):
        """初始化访问控制器"""
        self.user_roles: Dict[str, str] = {}
        self.doc_permissions: Dict[str, Dict[str, Set[Permission]]] = {}
        # doc_permissions[doc_id][user_id] = {Permission.READ, ...}
    
    def assign_role(self, user_id: str, role_name: str):
        """
        为用户分配角色
        
        Args:
            user_id: 用户 ID
            role_name: 角色名称
        """
        if role_name not in ROLES:
            raise ValueError(f"未知角色: {role_name}")
        
        self.user_roles[user_id] = role_name
        print(f"用户 {user_id} 分配角色: {role_name}")
    
    def set_document_permission(
        self,
        doc_id: str,
        user_id: str,
        permissions: Set[Permission]
    ):
        """
        设置文档级权限
        
        Args:
            doc_id: 文档 ID
            user_id: 用户 ID
            permissions: 权限集合
        """
        if doc_id not in self.doc_permissions:
            self.doc_permissions[doc_id] = {}
        
        self.doc_permissions[doc_id][user_id] = permissions
    
    def check_permission(
        self,
        user_id: str,
        permission: Permission,
        doc_id: str = None
    ) -> bool:
        """
        检查用户权限
        
        Args:
            user_id: 用户 ID
            permission: 要检查的权限
            doc_id: 文档 ID（可选，用于文档级权限）
        
        Returns:
            是否有权限
        """
        # 1. 检查角色权限
        role_name = self.user_roles.get(user_id)
        if not role_name:
            return False
        
        role = ROLES[role_name]
        if not role.has_permission(permission):
            return False
        
        # 2. 检查文档级权限（如果指定了文档）
        if doc_id:
            doc_perms = self.doc_permissions.get(doc_id, {})
            user_doc_perms = doc_perms.get(user_id)
            
            if user_doc_perms is not None:
                return permission in user_doc_perms
        
        return True
    
    def filter_accessible_docs(
        self,
        user_id: str,
        permission: Permission,
        doc_ids: List[str]
    ) -> List[str]:
        """
        过滤用户可访问的文档
        
        Args:
            user_id: 用户 ID
            permission: 权限
            doc_ids: 文档 ID 列表
        
        Returns:
            可访问的文档 ID 列表
        """
        return [
            doc_id for doc_id in doc_ids
            if self.check_permission(user_id, permission, doc_id)
        ]

def require_permission(permission: Permission):
    """
    权限检查装饰器
    
    用于保护函数/方法
    """
    def decorator(func):
        @wraps(func)
        def wrapper(self, user_id: str, *args, **kwargs):
            ac = getattr(self, 'access_control', None)
            if ac and not ac.check_permission(user_id, permission):
                raise PermissionError(
                    f"用户 {user_id} 没有 {permission.value} 权限"
                )
            return func(self, user_id, *args, **kwargs)
        return wrapper
    return decorator

# 使用示例
"""
ac = AccessControl()
```

```python
"""
数据安全模块

实现数据加密、脱敏和审计
"""

from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64
import hashlib
import re

class DataEncryption:
    """数据加密器"""
    
    def __init__(self, master_key: str):
        """
        初始化加密器
        
        Args:
            master_key: 主密钥
        """
        self.master_key = master_key
        self.fernet = self._create_fernet(master_key)
    
    def _create_fernet(self, key: str) -> Fernet:
        """创建 Fernet 实例"""
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=b'salt',  # 实际使用应随机生成并存储
            iterations=100000,
        )
        key_bytes = base64.urlsafe_b64encode(kdf.derive(key.encode()))
        return Fernet(key_bytes)
    
    def encrypt(self, data: str) -> str:
        """加密数据"""
        return self.fernet.encrypt(data.encode()).decode()
    
    def decrypt(self, encrypted_data: str) -> str:
        """解密数据"""
        return self.fernet.decrypt(encrypted_data.encode()).decode()
    
    def hash_id(self, doc_id: str) -> str:
        """哈希 ID（用于匿名化）"""
        return hashlib.sha256(doc_id.encode()).hexdigest()[:16]

class DataMasking:
    """数据脱敏器"""
    
    @staticmethod
    def mask_phone(phone: str) -> str:
        """手机号脱敏"""
        return re.sub(r'(\d{3})\d{4}(\d{4})', r'\1****\2', phone)
    
    @staticmethod
    def mask_email(email: str) -> str:
        """邮箱脱敏"""
        return re.sub(r'(\w{2})\w+(@\w+)', r'\1***\2', email)
    
    @staticmethod
    def mask_id_card(id_card: str) -> str:
        """身份证号脱敏"""
        return id_card[:6] + "********" + id_card[-4:]
    
    @staticmethod
    def mask_sensitive_content(content: str) -> str:
        """
        内容脱敏
        
        检测并脱敏敏感信息
        """
        # 手机号
        content = re.sub(
            r'1[3-9]\d{9}',
            lambda m: DataMasking.mask_phone(m.group()),
            content
        )
        
        # 邮箱
        content = re.sub(
            r'[\w.-]+@[\w.-]+\.\w+',
            lambda m: DataMasking.mask_email(m.group()),
            content
        )
        
        # 身份证号
        content = re.sub(
            r'\d{17}[\dXx]|\d{15}',
            lambda m: DataMasking.mask_id_card(m.group()),
            content
        )
        
        return content

class AuditLogger:
    """审计日志器"""
    
    def __init__(self, log_file: str = "audit.log"):
        """
        初始化审计日志器
        
        Args:
            log_file: 日志文件路径
        """
        self.log_file = log_file
    
    def log_access(
        self,
        user_id: str,
        action: str,
        resource: str,
        status: str = "success",
        details: Dict = None
    ):
        """
        记录访问日志
        
        Args:
            user_id: 用户 ID
            action: 操作类型（read/write/delete/search）
            resource: 资源标识
            status: 操作状态
            details: 额外详情
        """
        import json
        from datetime import datetime
        
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "user_id": user_id,
            "action": action,
            "resource": resource,
            "status": status,
            "details": details or {}
        }
        
        with open(self.log_file, 'a', encoding='utf-8') as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + '\n')
    
    def query_logs(
        self,
        user_id: str = None,
        action: str = None,
        start_time: datetime = None,
        end_time: datetime = None
    ) -> List[Dict]:
        """
        查询审计日志
        
        Args:
            user_id: 过滤用户
            action: 过滤操作
            start_time: 开始时间
            end_time: 结束时间
        
        Returns:
            日志条目列表
        """
        import json
        
        logs = []
        
        try:
            with open(self.log_file, 'r', encoding='utf-8') as f:
                for line in f:
                    entry = json.loads(line.strip())
                    
                    # 应用过滤条件
                    if user_id and entry["user_id"] != user_id:
                        continue
                    if action and entry["action"] != action:
                        continue
                    
                    entry_time = datetime.fromisoformat(entry["timestamp"])
                    if start_time and entry_time < start_time:
                        continue
                    if end_time and entry_time > end_time:
                        continue
                    
                    logs.append(entry)
        except FileNotFoundError:
            pass
        
        return logs
```

```python
"""
系统监控模块

实现性能指标采集和告警
"""

import time
from collections import deque
from typing import Dict, List
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class Metric:
    """指标数据"""
    name: str
    value: float
    timestamp: datetime = field(default_factory=datetime.now)
    labels: Dict = field(default_factory=dict)

class MetricsCollector:
    """指标采集器"""
    
    def __init__(self, max_history: int = 10000):
        """
        初始化指标采集器
        
        Args:
            max_history: 最大历史记录数
        """
        self.metrics: Dict[str, deque] = {}
        self.max_history = max_history
    
    def record(self, metric: Metric):
        """记录指标"""
        if metric.name not in self.metrics:
            self.metrics[metric.name] = deque(maxlen=self.max_history)
        
        self.metrics[metric.name].append(metric)
    
    def get_metrics(
        self,
        name: str,
        start_time: datetime = None,
        end_time: datetime = None
    ) -> List[Metric]:
        """获取指标"""
        if name not in self.metrics:
            return []
        
        metrics = list(self.metrics[name])
        
        if start_time:
            metrics = [m for m in metrics if m.timestamp >= start_time]
        if end_time:
            metrics = [m for m in metrics if m.timestamp <= end_time]
        
        return metrics
    
    def get_stats(self, name: str) -> Dict:
        """获取统计信息"""
        metrics = self.get_metrics(name)
        
        if not metrics:
            return {}
        
        values = [m.value for m in metrics]
        
        return {
            "count": len(values),
            "avg": sum(values) / len(values),
            "min": min(values),
            "max": max(values),
            "p50": self._percentile(values, 50),
            "p95": self._percentile(values, 95),
            "p99": self._percentile(values, 99)
        }
    
    @staticmethod
    def _percentile(values: List[float], p: float) -> float:
        """计算百分位数"""
        sorted_values = sorted(values)
        k = (len(sorted_values) - 1) * p / 100
        f = int(k)
        c = f + 1 if f + 1 < len(sorted_values) else f
        
        if f == c:
            return sorted_values[f]
        
        return sorted_values[f] * (c - k) + sorted_values[c] * (k - f)

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self):
        """初始化性能监控器"""
        self.collector = MetricsCollector()
    
    def timeit(self, operation_name: str):
        """
        耗时统计装饰器
        
        用法:
            @monitor.timeit("search")
            def search(query):
                ...
        """
        def decorator(func):
            def wrapper(*args, **kwargs):
                start = time.time()
                try:
                    result = func(*args, **kwargs)
                    status = "success"
                    return result
                except Exception as e:
                    status = "error"
                    raise
                finally:
                    duration = time.time() - start
                    self.collector.record(Metric(
                        name=f"{operation_name}_duration",
                        value=duration,
                        labels={"status": status}
                    ))
            return wrapper
        return decorator
    
    def record_counter(self, name: str, value: int = 1, labels: Dict = None):
        """记录计数器"""
        self.collector.record(Metric(
            name=name,
            value=value,
            labels=labels or {}
        ))
    
    def record_gauge(self, name: str, value: float, labels: Dict = None):
        """记录仪表盘值"""
        self.collector.record(Metric(
            name=name,
            value=value,
            labels=labels or {}
        ))

class AlertManager:
    """告警管理器"""
    
    def __init__(self):
        """初始化告警管理器"""
        self.rules: List[Dict] = []
        self.alerts: deque = deque(maxlen=1000)
    
    def add_rule(
        self,
        metric_name: str,
        threshold: float,
        operator: str = ">",
        duration: int = 300
    ):
        """
        添加告警规则
        
        Args:
            metric_name: 指标名称
            threshold: 阈值
            operator: 比较运算符（>, <, ==）
            duration: 持续时间（秒）
        """
        self.rules.append({
            "metric_name": metric_name,
            "threshold": threshold,
            "operator": operator,
            "duration": duration,
            "active_since": None
        })
    
    def check_alerts(self, collector: MetricsCollector):
        """检查告警条件"""
        for rule in self.rules:
            stats = collector.get_stats(rule["metric_name"])
            
            if not stats:
                continue
            
            current_value = stats.get("avg", 0)
            triggered = self._evaluate_rule(rule, current_value)
            
            if triggered:
                if rule["active_since"] is None:
                    rule["active_since"] = time.time()
                elif time.time() - rule["active_since"] >= rule["duration"]:
                    self._trigger_alert(rule, current_value)
            else:
                rule["active_since"] = None
    
    def _evaluate_rule(self, rule: Dict, value: float) -> bool:
        """评估规则"""
        op = rule["operator"]
        threshold = rule["threshold"]
        
        if op == ">":
            return value > threshold
        elif op == "<":
            return value < threshold
        elif op == "==":
            return value == threshold
        
        return False
    
    def _trigger_alert(self, rule: Dict, value: float):
        """触发告警"""
        alert = {
            "timestamp": datetime.now().isoformat(),
            "metric": rule["metric_name"],
            "threshold": rule["threshold"],
            "current_value": value,
            "message": f"{rule['metric_name']} 当前值 {value:.2f} 超过阈值 {rule['threshold']}"
        }
        
        self.alerts.append(alert)
        print(f"[ALERT] {alert['message']}")

# 使用示例
"""
monitor = PerformanceMonitor()

# 装饰器方式监控
@monitor.timeit("search")

    time.sleep(0.1)  # 模拟搜索

monitor.record_gauge("index_size", 1000000)

alert_manager.check_alerts(monitor.collector)

print(f"搜索耗时统计: {stats}")
"""
```
